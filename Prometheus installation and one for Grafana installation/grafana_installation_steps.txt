# Grafana Installation & Setup

# 1. Add Grafana APT repository
sudo apt-get install -y software-properties-common
sudo add-apt-repository "deb https://packages.grafana.com/oss/deb stable main"
wget -q -O - https://packages.grafana.com/gpg.key | sudo apt-key add -

# 2. Update and install Grafana
sudo apt-get update
sudo apt-get install grafana -y

# 3. Start Grafana service
sudo systemctl start grafana-server
sudo systemctl enable grafana-server

# 4. Access Grafana UI
http://localhost:3000

# 5. Default login
Username: admin
Password: admin

# 6. Configure Prometheus as datasource
- Go to Grafana UI → Configuration → Data Sources
- Add Prometheus
- URL: http://localhost:9090

# 7. Import dashboards
- Go to Grafana UI → Dashboards → Import
- Use dashboard IDs from Grafana.com (e.g., PostgreSQL Overview, MySQL Overview)
